def check_for_microservices(folder_path):
    """Recursively checks for Spring Boot microservices by looking for pom.xml files."""
    for root, dirs, files in os.walk(folder_path):
        # Check if the current folder contains a 'pom.xml'
        if 'pom.xml' in files:
            print(f"Found pom.xml in folder: {root}")
