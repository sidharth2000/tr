ngAfterViewInit() {
    // Add event listener for the copy buttons after the view is initialized
    const copyButtons = document.querySelectorAll('.copy-button');
    copyButtons.forEach(button => {
      button.addEventListener('click', this.copyToClipboard);
    });
  }

  // Copy to clipboard function
  copyToClipboard(event: Event): void {
    const button = event.target as HTMLButtonElement;
    const codeBlock = button.nextElementSibling as HTMLElement;
    const codeText = codeBlock.innerText;

    navigator.clipboard.writeText(codeText)
      .then(() => {
        alert('Code copied to clipboard!');
      })
      .catch(err => {
        console.error('Failed to copy text: ', err);
      });
  }
