modules_match = re.search(r"^MODULES\s*=\s*(.*)", content, re.MULTILINE)
app_match = re.search(r"^APP\s*=\s*(.*)", content, re.MULTILINE)

# Extract and split MODULES
modules = modules_match.group(1).split() if modules_match else []
app = app_match.group(1).strip() if app_match else ""

# Print results
print("Modules:", modules)
print("App:", app)
