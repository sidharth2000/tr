constructor(private renderer: Renderer2, private el: ElementRef) {}

  ngAfterViewInit(): void {
    this.bindCopyButtons();
  }

  bindCopyButtons(): void {
    // Find all buttons with id 'copy' (preferably use class for multiple buttons with the same name)
    const buttons = this.el.nativeElement.querySelectorAll('#copy');
    
    // Loop through each button and add a click listener
    buttons.forEach((button: HTMLElement) => {
      this.renderer.listen(button, 'click', () => this.copyContent(button));
    });
  }

  copyContent(button: HTMLElement): void {
    // Example content to copy (could be any content or text related to the button)
    const contentToCopy = `Content from: ${button.innerText}`;

    // Create a temporary textarea element to copy the content to clipboard
    const textArea = document.createElement('textarea');
    textArea.value = contentToCopy;
    document.body.appendChild(textArea);
    textArea.select();
    document.execCommand('copy'); // Copy the text to clipboard
    document.body.removeChild(textArea);

    // Optionally, show a message or change the button text to indicate success
    alert(`Copied: ${contentToCopy}`);
  }
